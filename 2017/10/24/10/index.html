    有个目录，里面是你自己写过的程序，统计一下你写过多少行代码。包括空行和注释，但是要分别列出来。

    解析

    没啥好解析的，看代码就能看懂。先找出所有文件，然后对每个文件进行判断

    代码

    #coding:utf-8
import os
def get_file(path): #遍历该文件夹下的所有文件
    files = os.listdir(path)
    paths = []
    for f in files:
        f_path = path + '\\' + f
        if os.path.isfile(f_path): #判断是否是文件
            if f.split('.')[1] == 'py': #判断是否为.py文件
                paths.append(f)
        elif os.path.isdir(f_path): #判断是否是文件夹
            paths += get_file(f_path)
    return paths #返回所有的文件地址
def run():
    files = get_file('.')
    code,comments,blank = 0,0,0
    for file in files:
        with open(file,'r',encoding='utf-8') as f:
            for line in f:
                line = line.strip() #去掉前后空格
                code += 1
                if line == '': #判断是否是空白行
                    blank += 1
                elif line[0] == '#': #判断是否是注释
                    comments += 1
    print('code:%d  comments:%d    blank:%d'  %(code,comments,blank))

if __name__ == '__main__':
    run()
